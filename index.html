myWebView.setWebViewClient(new WebViewClient() { @Override public boolean shouldOverrideUrlLoading(WebView view, String url) { if (url.startsWith("intent://")) { try { Intent intent = Intent.parseUri(url, Intent.URI_INTENT_SCHEME); if (intent != null) { view.getContext().startActivity(intent); return true; } } catch (URISyntaxException e) { e.printStackTrace(); } } return false; } });
